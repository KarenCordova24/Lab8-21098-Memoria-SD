//************************************************************************************************

void setup() {
  SPI.setModule(0);
  Serial.begin(115200);
  pinMode(pushM, INPUT_PULLUP);
  pinMode(LEDR, OUTPUT);
  pinMode(LEDV, OUTPUT);
  pinMode(12, OUTPUT);

  if (!SD.begin(12)) {
    Serial.println("¡Inicialización fallida!");
    return;
  }
  Serial.println("¡Inicialización completa!");
}
